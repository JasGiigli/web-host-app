Dynamic Styling with JavaScript:

**Changing the color on cliking the button** 

            <code>
                <!DOCTYPE html>
                <html lang="en">
                <head>
                    <meta charset="UTF-8" />
                    <meta name="viewport" content="width=device-width, 
                    initial-scale=1.0" />
                    <title>Document</title>
                </head>
                <body>
                    <button id="changeColorButton" onclick="
                    changeColor()">Change Color</button>

                    //JavaScript code
                    <script>
                    function changeColor() {
                        var element = document
                        .getElementById('changeColorButton');
                        element.style.backgroundColor = 'blue';
                        element.style.color = 'white';
                        element.textContent = 'Color Changed!';
                    }
                </script>
                </body>
                </html>
            </code>

Toggling Element Visibility:

**Click the button to toggle visibility** 

              <code>
                  <!DOCTYPE html>
                  <html lang="en">
                  <head>
                      <meta charset="UTF-8" />
                      <meta name="viewport" content="width=device-width,
                       initial-scale=1.0" />
                      <title>Document</title>
                  </head>
                  <body>
                      <div id="elementToToggle">
                          This element can be toggled!
                      </div>
                      <button onclick="toggleVisibility()">
                      Toggle Element</button>
      
                      <script>
                          function toggleVisibility() {
                              var element = document
                              .getElementById('elementToToggle');
                              if (element.style.display === 'none') {
                                  element.style.display = 'block';
                              } else {
                                  element.style.display = 'none';
                              }
                          }
                      </script>
                  </body>
                  </html>
              </code>
